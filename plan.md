## 第一阶段：方向调整（桌面端优先）
由于 IPv6 + 反代链路在 Docker/WSL2 网络下不稳定，项目主路线改为 **Windows 桌面应用**，避免容器网络干扰。

主技术栈：
- GUI：PySide6 + QWidget
- 设计风格：Fluent 风格（QFluentWidgets 或同等组件方案）
- 核心引擎：沿用现有 Python 抽帧逻辑（ffmpeg / ffprobe）
- 数据存储：SQLite（卡片配置、任务记录）
- 调度：APScheduler（每卡片独立 Cron）

说明：
- Docker + Web 方案降级为“可选分支”，不再是主交付。
- 你当前确认满意的页面结构与交互（卡片管理、新增区、日志区、主题切换）将完整保留为桌面版设计规范。

```python
#之前的能用的核心代码 供参考
import os
import subprocess
import time
import threading
import argparse
from pathlib import Path
from concurrent.futures import ThreadPoolExecutor

try:
    import psutil
except ImportError:
    psutil = None

try:
    from tqdm import tqdm
except ImportError:
    tqdm = None

class TrafficStats:
    def __init__(self):
        self.total_download = 0 
        self.total_output = 0   
        self.success_count = 0
        self.fail_count = 0
        self.start_time = time.time()
        self._lock = threading.Lock()

    def add_process_traffic(self, proc_read_bytes):
        with self._lock:
            self.total_download += proc_read_bytes

    def add_output_size(self, size):
        with self._lock:
            self.total_output += size

    def inc_success(self):
        with self._lock:
            self.success_count += 1

    def inc_fail(self):
        with self._lock:
            self.fail_count += 1

    def report(self):
        duration = time.time() - self.start_time
        print("\n" + "="*40)
        print("         流量与任务统计报告")
        print("-" * 40)
        print(f"处理时长:     {duration:.2f} 秒")
        print(f"成功/失败:     {self.success_count} / {self.fail_count}")
        print(f"生成文件总量: {self.total_output / 1024:.2f} KB")
        
        if psutil:
            print(f"累计下载流量: {self.total_download / 1024 / 1024:.2f} MB")
            print("注: '累计下载流量' 基于网卡统计，包含所有并发任务的网络开销。")
        else:
            print("注: 未安装 psutil, 无法统计详细流量。")
        print("="*40 + "\n")

class QPSLimiter:
    def __init__(self, qps=2):
        self.interval = 1.0 / qps
        self.last_shot = 0
        self._lock = threading.Lock()

    def wait(self):
        with self._lock:
            now = time.time()
            wait_time = self.interval - (now - self.last_shot)
            if wait_time > 0:
                time.sleep(wait_time)
            self.last_shot = time.time()

stats = TrafficStats()
limiter = QPSLimiter(qps=2)

def get_video_duration(url):
    """获取视频时长 (秒)"""
    try:
        cmd = [
            'ffprobe', 
            '-v', 'error', 
            '-show_entries', 'format=duration', 
            '-of', 'default=noprint_wrappers=1:nokey=1', 
            url
        ]
        
        start_recv = psutil.net_io_counters().bytes_recv if psutil else 0
        result = subprocess.run(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True, timeout=20)
        if psutil:
            stats.add_process_traffic(max(0, psutil.net_io_counters().bytes_recv - start_recv))

        if result.returncode == 0:
            return float(result.stdout.strip())
    except Exception:
        pass
    return None

def extract_frame(url, output_path, timestamp):
    """提取指定时间点的帧"""
    try:
        cmd = [
            'ffmpeg', '-y',
            '-ss', str(round(timestamp, 2)),
            '-i', url,
            '-map', '0:v:0',
            '-frames:v', '1',
            '-q:v', '3',
            '-f', 'image2',
            str(output_path)
        ]
        start_recv = psutil.net_io_counters().bytes_recv if psutil else 0
        result = subprocess.run(cmd, stdout=subprocess.DEVNULL, stderr=subprocess.PIPE, timeout=60)
        if psutil:
            stats.add_process_traffic(max(0, psutil.net_io_counters().bytes_recv - start_recv))

        if result.returncode == 0 and output_path.exists():
            stats.add_output_size(output_path.stat().st_size)
            return True
    except Exception:
        pass
    return False

def generate_nfo(strm_path, duration):
    """生成基础 NFO 文件"""
    nfo_path = strm_path.with_suffix('.nfo')
    title = strm_path.stem # 比如 Addiction_2160P.mp4
    runtime = int(duration / 60) if duration else 0
    
    content = f"""<?xml version="1.0" encoding="utf-8" standalone="yes"?>
<movie>
  <title>{title}</title>
  <originaltitle>{title}</originaltitle>
  <runtime>{runtime}</runtime>
  <plot>Auto generated by STRM Scraper</plot>
</movie>"""
    
    try:
        with open(nfo_path, 'w', encoding='utf-8') as f:
            f.write(content)
        return True
    except:
        return False

def process_single_strm(strm_path, args):
    """处理单个 STRM 的核心逻辑"""
    poster_path = strm_path.with_suffix('.jpg')
    fanart_path = Path(str(strm_path).replace('.strm', '-fanart.jpg'))
    nfo_path = strm_path.with_suffix('.nfo')

    # 查重逻辑（增量模式）
    if not args.full:
        need_poster = args.poster and not poster_path.exists()
        need_fanart = args.fanart and not fanart_path.exists()
        need_nfo = args.nfo and not nfo_path.exists()
        if not (need_poster or need_fanart or need_nfo):
            return "skipped"
    else:
        # 全量模式：如果存在则删除
        for p in [poster_path, fanart_path, nfo_path]:
            if p.exists(): p.unlink()

    try:
        with open(strm_path, 'r', encoding='utf-8') as f:
            url = next((line.strip() for line in f if line.strip() and not line.startswith('#')), "")
        
        if not url: return "failed"

        # QPS 限制请求节点
        limiter.wait()
        
        duration = get_video_duration(url)
        if not duration: return "failed"

        success = True
        if args.poster:
            success &= extract_frame(url, poster_path, duration * 0.1) # 10% 进度
        
        if args.fanart:
            success &= extract_frame(url, fanart_path, duration * 0.5) # 50% 进度作为背景图
            
        if args.nfo:
            success &= generate_nfo(strm_path, duration)

        if success:
            stats.inc_success()
            return "success"
        else:
            stats.inc_fail()
            return "failed"
            
    except Exception:
        stats.inc_fail()
        return "failed"

def main():
    parser = argparse.ArgumentParser(description="STRM 媒体搜刮工具")
    parser.add_argument("--path", default=r"X:\Docker\Qmediasync\strm", help="STRM 根目录")
    parser.add_argument("--full", action="store_true", help="全量模式 (覆盖现有文件)")
    parser.add_argument("--threads", type=int, default=4, help="线程数 (建议 2-4)")
    parser.add_argument("--poster", action="store_true", default=True, help="抓取封面 (10%)")
    parser.add_argument("--fanart", action="store_true", help="抓取背景图 (50%)")
    parser.add_argument("--nfo", action="store_true", help="生成 NFO 文件")
    args = parser.parse_args()

    base_dir = Path(args.path)
    if not base_dir.exists():
        print(f"路径不存在: {base_dir}")
        return

    strm_files = list(base_dir.glob("**/*.strm"))
    total = len(strm_files)
    print(f"模式: {'全量' if args.full else '增量'} | 线程: {args.threads} | 任务总数: {total}")

    # 使用进度条
    pbar = tqdm(total=total, desc="搜刮进度") if tqdm else None

    with ThreadPoolExecutor(max_workers=args.threads) as executor:
        futures = {executor.submit(process_single_strm, f, args): f for f in strm_files}
        
        for future in futures:
            res = future.result()
            if pbar:
                pbar.update(1)
                # 可选：显示当前状态
                # pbar.set_postfix({"last": futures[future].name})

    if pbar: pbar.close()
    stats.report()

if __name__ == "__main__":
    main()
```

## 第二阶段：桌面应用核心架构
### 1) 模块拆分
- `core/runner.py`：任务调度与执行入口（全量/增量、单卡片/全部卡片）
- `core/scraper.py`：抽帧与 NFO 逻辑（复用并增强现有代码）
- `core/network_guard.py`：网络健壮性（重试、超时、IPv4 优先、失败分类）
- `core/storage.py`：SQLite 读写（配置卡片、统计、日志）
- `ui/main_window.py`：主界面（Dashboard + 日志 + 卡片区）
- `ui/dialogs/profile_editor.py`：卡片编辑弹窗

### 2) 数据模型（目录卡片）
每张卡片字段：
- 卡片名称
- 目录路径（可浏览并选择，不靠手输）
- 线程数
- Cron 表达式（默认 `0 2 * * *`）
- 启用卡片
- 启用定时任务
- 生成 NFO
- 覆盖已存在文件
- 生成海报
- 生成横图
- 海报抽帧百分比
- 横图抽帧百分比

### 3) 运行模式
- 全局运行按钮：增量（高亮）/全量（双确认）
- 单卡片运行：使用界面外部统一模式切换（默认增量）
